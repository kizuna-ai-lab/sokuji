python3 pack.py kitten                  # Kitten Nano EN v0.2

python3 pack.py matcha-en               # Matcha EN + vocos vocoder

python3 pack.py zipvoice-distill-fp32   # ZipVoice fp32 + vocos
python3 pack.py pocket-int8             # Pocket TTS int8
python3 pack.py pocket                  # Pocket TTS fp32
python3 pack.py coqui-en                # Coqui EN (LJSpeech)
python3 pack.py mimic3-ko               # Mimic3 Korean
python3 pack.py mms-eng                 # MMS English
python3 pack.py piper-en                # Piper EN (fp32 + int8)
python3 pack.py all                     # all models, all variants

python3 -m http.server 8080
